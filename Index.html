const fetch = require("node-fetch"); exports.handler = async (event) => { const body = JSON.parse(event.body); const prompt = body.message; const response = await fetch("https://openrouter.ai/api/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${process.env.OPENROUTER_API_KEY}`, "Content-Type": "application/json", }, body: JSON.stringify({ model: "openai/gpt-3.5-turbo", messages: [{ role: "user", content: prompt }], }), }); const data = await response.json(); return { statusCode: 200, body: JSON.stringify({ reply: data.choices[0].message.content }) }; };
